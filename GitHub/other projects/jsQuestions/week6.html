$("#q6button").on("click", function(){
                    function range(start,end){
                        var arr = [];
                        for(start; start <= end; start++){
                            arr.push(start);
                        }
                        return arr;
                    };
                    var rangeResult = range(2,6);
                    $("#q6result").html(rangeResult.join());
                });
$("#q7button").on("click", function(){
                    var arr = [1,2,3,4,5,6];
                    var total = 0;
                    for(var i in arr){
                        total += arr[i];
                    }                    
                    $("#q7result").html(total);
                });
$("#q8button").on("click",function(){
                    function range(end,start){
                        var arr = [];
                        for(end; end >= start; end--){
                            arr.push(end);
                        }
                        return arr;
                    };
                    var reverseArray = range(6,2);
                    $("#q8result").html(reverseArray.join());
                });
